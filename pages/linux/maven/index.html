搭建Nexus（maven私服）

# 文档

  • Nexus的下载地址：https://www.sonatype.com/download-oss-sonatype
  • 官方安装文档：https://help.sonatype.com/repomanager3/installation/installation-methods

# CentOS7安装maven私服

下载地址：https://pan.bjtxra.com/#routeview/s/default/0/0/share$xvPsHTCH

tar xvf nexus-3.25.1-04-unix.tar.gz -C /usr/src/
# 修改端口，默认是8081
vi /usr/src/nexus-3.25.1-04/etc/nexus-default.properties
 # 开放8081端口
firewall-cmd --zone=public --add-port=8081/tcp --permanent & firewall-cmd --reload
# 启动nexus
cd /usr/src/nexus-3.25.1-04/bin && ./nexus start
# 登录 192.16.18.99:8081
# 初始话admin的密码存在 /usr/src/sonatype-work/nexus3/admin.password 重置以后，改文件消失

# 设置开机自启（推荐）

https://help.sonatype.com/repomanager3/installation/system-requirements

cat >> /usr/lib/systemd/system/nexus3.service << EOF
[Unit]
Description=nexus service
After=network.target

[Service]
Type=forking
LimitNOFILE=65536
Environment="JAVA_HOME=/usr/local/lib/jdk1.8.0_261/"
ExecStart=/usr/src/nexus-3.25.1-04/bin/nexus start
ExecReload=/usr/src/nexus-3.25.1-04/bin/nexus restart
ExecStop=/usr/src/nexus-3.25.1-04/bin/nexus stop
User=root
Restart=on-abort

[Install]
WantedBy=multi-user.target
EOF

# 设置开机自启

vim /etc/init.d/nexus
1
#!/bin/bash    
#chkconfig:2345 20 90    
#description:nexus3    
#processname:nexus3    
export JAVA_HOME=/usr/local/lib/jdk1.8.0_261
case $1 in    
        start) su root /usr/src/nexus-3.25.1-04/bin/nexus start;;    
        stop) su root /usr/src/nexus-3.25.1-04/bin/nexus stop;;    
        status) su root /usr/src/nexus-3.25.1-04/bin/nexus status;;    
        restart) su root /usr/src/nexus-3.25.1-04/bin/nexus restart;;    
        dump ) su root /usr/src/nexus-3.25.1-04/bin/nexus dump ;; 
        console ) su root /usr/src/nexus-3.25.1-04/bin/nexus console ;;          
        *) echo "require console | start | stop | restart | status | dump " ;;    
esac

# 全局配置下载依赖(即项目pom无需配置)

  • 在maven的setting.xml文件中配置私服配置，这种方式配置后所有本地使用该配置的maven项目的pom文件都无需配置私服下载相关配置
    <profiles>
    <profile>
        <id>mymaven</id>
            <repositories>
            <!-- 私有库地址-->
            <repository>
            <id>nexus</id>
            <url>http://192.16.18.99:8081/repository/maven-public/</url>
            <releases>
                <enabled>true</enabled>
            </releases>
            <snapshots>
                <enabled>true</enabled>
            </snapshots>
            </repository>
        </repositories>      
        <pluginRepositories>
            <!--插件库地址-->
            <pluginRepository>
            <id>nexus</id>
            <url>http://192.16.18.999:8081/repository/maven-public/</url>
            <releases>
                <enabled>true</enabled>
            </releases>
            <snapshots>
                <enabled>true</enabled>
            </snapshots>
            </pluginRepository>
        </pluginRepositories>
    </profile>
</profiles>

# 单独项目下载依赖(即项目pom文件中配置)

这种配置是修改单个项目的pom文件，无需修改maven的setting配置(尽管如此说，但是如果setting.xml中配置了mirror标签，并且mirrorOf为*或者私服地址，还是会被拦截，被代理指向代理地址)

<repositories>
  <repository>
    <id>nexus</id>
    <url>http://192.16.18.99:8081/repository/maven-public/</url>
    <releases>
      <enabled>true</enabled>
    </releases>
    <snapshots>
    <enabled>true</enabled>
    </snapshots>
  </repository>
</repositories>

# 本地maaven开发的项目上传配置

    <server>
        <id>nexus-releases</id>
        <username>admin</username>
        <password>admin</password>
    </server>
    <server>
        <id>nexus-snapshots</id>
        <username>admin</username>
        <password>admin</password>
    </server>
